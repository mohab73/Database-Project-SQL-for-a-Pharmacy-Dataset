CREATE TABLE SUPPLIERS(
ID NUMBER(15) PRIMARY KEY,
CONTACT_INFO VARCHAR2(30) NOT NULL,
PLACE_INFO VARCHAR2(25) NOT NULL,
PHONE_NUMBER NUMBER(11) UNIQUE
);

CREATE TABLE ITEM(
BARCODE NUMBER(15) PRIMARY KEY,
PRICE NUMBER(5) NOT NULL,
FORMULA VARCHAR2(20) NOT NULL,
EXP DATE NOT NULL,
SUPPLYING_COMPANY VARCHAR2(15) NOT NULL
);

CREATE TABLE SENDING(
ID NUMBER(15),
BARCODE NUMBER(15),
PRIMARY KEY (ID , BARCODE),
FOREIGN KEY (ID) REFERENCES SUPPLIERS (ID),
FOREIGN KEY (BARCODE) REFERENCES ITEM (BARCODE)
);

CREATE TABLE MEDICINE(
PARTS NUMBER(2) NOT NULL,
AVAILABILITY VARCHAR2(3) NOT NULL,
BARCODE NUMBER(15) PRIMARY KEY,
FOREIGN KEY (BARCODE) REFERENCES ITEM (BARCODE) ON DELETE SET NULL
);

CREATE TABLE PERSONAL_CARE_PRODUCTS(
BRAND VARCHAR2(12) NOT NULL,
BARCODE NUMBER(15) PRIMARY KEY,
FOREIGN KEY (BARCODE) REFERENCES ITEM (BARCODE)
);

CREATE TABLE employees(
id number(10) PRIMARY KEY,
address varchar2(50),
salary number(10),
bonus number(10),
full_name varchar2(15),
attendance date
);

CREATE TABLE sales_movement(
  ssn number(10) NOT NULL,
  sales_id number(10) NOT NULL,
  sell number(20),
  revenue number(20),
  loss number(20),
  average_sales number(20),
  average_revenue number(20),
  average_loss number(20),
  prescription varchar2(200),
  PRIMARY KEY(ssn),
  FOREIGN KEY (sales_id) REFERENCES employees (id)
);

CREATE TABLE sold(
  ssn number(10) NOT NULL ,
  barcode number(10) NOT NULL ,
  PRIMARY KEY (ssn,barcode),
  FOREIGN KEY (ssn) REFERENCES sales_movement (ssn)
);
